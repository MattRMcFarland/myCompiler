PRINT ( STRING ) ;

TYPEINT ID = INT ;
TYPEINT ID [ ID ] = { INT } ;

TYPEINT ID ;
FOR ( TYPEINT ID = INT ; ID < ID ; ID INCREMENT ) { 

ID = INT ;
WHILE ( ( ID * ID ) < ID ) {
ID [ ID * ID ] = ! ID [ ID * ID ] ;
ID INCREMENT ;
}
}

FOR ( TYPEINT ID = INT ; ID < ID ; ID INCREMENT ) {
IF ( ID [ ID ] ) {

} ELSE {

}
}

}
